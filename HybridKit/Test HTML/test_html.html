HybridKit

Demo

HybridKit is a simple, extensible messaging system for hybrid web <-> iOS/Android mobile apps. 
HybridKit.runCommand("alert", {title: "Title", message:"Message", cancel_button_title:"Cancel"})

HybridKit.runCommand("open_url", {url: "http://www.usepropeller.com"})

HybridKit.runCommand("set_title", {title: "Hello, HybridKit!"})

HybridKit.runCommand("set_scroll_enabled"), {enabled: false})

HybridKit.runCommand("set_scroll_enabled"), {enabled: true})

HybridKit.runCommand("set_background_color", {color: "red"})

HybridKit.runCommand("set_background_color", {color: "#887766"})

HybridKit.runCommand("deceleration_rate", {rate: "fast"})

HybridKit.runCommand("javascript", {javascript: "alert('Hey!')"})
